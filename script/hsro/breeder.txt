//NPC made by : HaVoC
prontera.gat,141,180,4	script	Rental NPC	726,{

mes "[Rental NPC]";
mes "Hi, here you can rent Carts, Falcons or Pecopecos.";
next;
menu "Rent a Cart",Cart,"Rent a Falcon",Falc,"Rent a Pecopeco",Peco,"Quit",Quit;


	Cart:
if(GetJob==5) goto Cart_Ok;
if(GetJob==10) goto Cart_Ok;
if(GetJob==18) goto Cart_Ok;
mes "[Rental NPC]";
mes "Sorry, but I only rent carts to Merchants, Blacksmiths and Alchimists.";
close;


	Cart_Ok:
	mes "Good job, here you go";
give_special 1;
	close;


	Falc:
if(GetJob==11) goto Falc_Ok;
mes "[Rental NPC]";
mes "Sorry, but I only rent falcons to Hunters.";
close;


	Falc_Ok:
	mes "Good job, here you go";
give_special 2;
	close;


	Peco:
if(GetJob==7) goto Peco_Ok;
if(GetJob==14) goto Peco_Ok;
mes "[Rental NPC]";
mes "Sorry, but I only lend Pecopecos to Knights and Crusaders.";
close;


	Peco_Ok:
	mes "Good job, here you go";
give_special 3;
	close;


	Quit:
	close;

}
